return; } console.log('Running in Desktop mode'); // Show desktop settings const desktopSettings = document.getElementById('desktop-settings'); if (desktopSettings) desktopSettings.style.display = 'block'; // Load saved settings try { const settings = await window.electronAPI.getSettings(); if (settings) { if (settings.downloadPath) { document.getElementById('desktop-download-path').textContent = settings.downloadPath; } if (settings.theme) { document.getElementById('theme-selector').value = settings.theme; if (settings.theme === 'light') document.body.classList.add('light-theme'); } if (settings.rememberSettings) { document.getElementById('remember-settings').checked = settings.rememberSettings; } if (settings.autoUpdate) { document.getElementById('auto-update').checked = settings.autoUpdate; } } } catch (e) { console.error('Error loading settings:', e); } // Get app version try { const version = await window.electronAPI.getAppVersion(); document.getElementById('app-version').textContent = `v${version}`; } catch (e) {} } // Add dropdown menu to user chip function addUserDropdown() { const userChip = document.querySelector('.user-chip'); if (!userChip) return; // Check if dropdown already exists if (userChip.parentElement.classList.contains('dropdown')) return; const dropdown = document.createElement('div'); dropdown.className = 'dropdown'; dropdown.style.position = 'relative'; const dropdownContent = document.createElement('div'); dropdownContent.className = 'dropdown-content'; dropdownContent.innerHTML = ` 🔄 Switch Account ℹ️ About & Info ⭐ Star on GitHub ⚙️ Preferences `; userChip.parentNode.insertBefore(dropdown, userChip); dropdown.appendChild(userChip); dropdown.appendChild(dropdownContent); } function openSettings() { switchTab('settings'); } // Account functions function switchAccount() { if (confirm('Switch to a different account? if (modal) modal.remove(); } } // Desktop settings functions function changeDownloadFolder() { if (window.electronAPI && window.electronAPI.selectDownloadFolder) { window.electronAPI.selectDownloadFolder().then(newPath => { if (newPath) { document.getElementById('desktop-download-path').textContent = newPath; toast('Download folder updated!', 'success'); } }); } else { toast('This feature is only available in desktop mode', 'info'); } } function changeTheme() { const theme = document.getElementById('theme-selector').value; if (theme === 'light') { document.body.classList.add('light-theme'); } else if (theme === 'dark') { document.body.classList.remove('light-theme'); } else if (theme === 'system') { const darkMode = window.matchMedia('(prefers-color-scheme: dark)').matches; if (darkMode) { document.body.classList.remove('light-theme'); } else { document.body.classList.add('light-theme'); } } if (window.electronAPI) { window.electronAPI.saveSettings({ theme: theme }); } } function toggleRememberSettings() { const remember = document.getElementById('remember-settings').checked; if (window.electronAPI) { window.electronAPI.saveSettings({ rememberSettings: remember }); } } function toggleAutoUpdate() { const autoUpdate = document.getElementById('auto-update').checked; if (window.electronAPI) { window.electronAPI.saveSettings({ autoUpdate: autoUpdate });
